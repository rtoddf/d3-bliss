// new way
var valueline = d3.line()
    .x(function(d) { return x(d.date); })
    .y(function(d) { return y(d.close); })
    .curve(d3.curveLinear)

// old way
var valueline = d3.svg.line()
	.x(function(d){ return xScale(d.x) })
	.y(function(d){ return yScale(d.y) })
